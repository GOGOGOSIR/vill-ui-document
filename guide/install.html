    # 安装

    npm install vill-ui --save

    # 全局引入

    import Vue from 'vue'
import VillUi from 'vill-ui'
import 'vill-ui/lib/common/style/index.css'

Vue.use(VillUi)

    # 按需引入

    1. 先安装babel-plugin-component
    npm install babel-plugin-component --save-dev
    1. 配置babel.config.js
    {
  "plugins": [
    [
      "component",
      {
        libraryName: "vill-ui",
        styleLibraryName: "common/style"
      }
    ]
  ]
}
    1. 按需引入
