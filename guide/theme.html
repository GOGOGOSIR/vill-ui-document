    # 新建一个theme.scss文件

    
$color-theme: skyblue;

$--font-path: '~vill-ui/src/common/style/iconfont';

@import "~vill-ui/src/common/style/index";

    # 在main.js中引入

    import Vue from 'vue'
import VillUi from 'vill-ui;
import "./theme.scss";

Vue.use(VillUi)
